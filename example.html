Advanced Examples

CSS Styling

inline-px {
    display: inline-block;
    filter: drop-shadow(2px 2px 4px rgba(0,0,0,0.5));
    transition: transform 0.2s;
}

inline-px:hover {
    transform: scale(1.1);
}

JavaScript API

// Create element dynamically
const sprite = document.createElement('inline-px');
sprite.setAttribute('data', '16x16:000...');
sprite.setAttribute('scale', '4');
document.body.appendChild(sprite);

// Update dynamically
sprite.setAttribute('scale', '8');
